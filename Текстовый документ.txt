<script>
    // ------------------- СЧЁТЧИК ДНЕЙ -------------------
    const START_DATE = new Date(2025, 0, 2);
    function getDaysTogether() {
        const today = new Date();
        today.setHours(0, 0, 0, 0);
        const start = new Date(START_DATE);
        start.setHours(0, 0, 0, 0);
        const diff = Math.floor((today - start) / (1000 * 60 * 60 * 24));
        return diff >= 0 ? diff : 0;
    }
    function updateCounter() {
        const days = getDaysTogether();
        document.getElementById('daysCounter').innerText = days;
        updateStats(days);
        return days;
    }
    function updateStats(days) {
        const hours = days * 24;
        const minutes = days * 24 * 60;
        const totalLifeDays = 80 * 365.25;
        let percent = (days / totalLifeDays) * 100;
        percent = Math.min(percent, 99.9);
        document.getElementById('statsBlock').innerHTML = `
            ⏱️ ${hours.toLocaleString()} часов вместе · 
            💖 ${minutes.toLocaleString()} минут<br>
            🌍 это ~ ${percent.toFixed(2)}% от целой жизни (80 лет)
        `;
    }

    // ------------------- СЕРДЕЧКИ -------------------
    let clickCount = 0, clickTimeout = null;
    function createFloatingHeart(x, y) {
        const heart = document.createElement('div');
        heart.className = 'floating-heart';
        heart.innerText = '❤️';
        heart.style.left = (x - 15) + 'px';
        heart.style.top = (y - 20) + 'px';
        heart.style.fontSize = (20 + Math.random() * 20) + 'px';
        document.body.appendChild(heart);
        setTimeout(() => heart.remove(), 1000);
    }
    function onCounterClick(e) {
        clickCount++;
        if (clickCount === 1) clickTimeout = setTimeout(() => clickCount = 0, 800);
        if (clickCount === 5) {
            const rect = e.target.getBoundingClientRect();
            const cx = rect.left + rect.width/2, cy = rect.top + rect.height/2;
            for (let i=0; i<12; i++) {
                setTimeout(() => {
                    const dx = (Math.random()-0.5)*40, dy = (Math.random()-0.5)*30;
                    createFloatingHeart(cx+dx, cy+dy);
                }, i*30);
            }
            if (navigator.vibrate) navigator.vibrate(100);
            clickCount = 0;
            clearTimeout(clickTimeout);
        }
    }
    document.querySelector('.heart')?.addEventListener('click', (e) => {
        const rect = e.target.getBoundingClientRect();
        const x = rect.left + rect.width/2, y = rect.top + rect.height/2;
        for(let i=0;i<4;i++) createFloatingHeart(x+(Math.random()-0.5)*30, y+(Math.random()-0.5)*20);
    });
    document.getElementById('daysCounter').addEventListener('click', onCounterClick);

    // ------------------- ФОТО ДНЯ (папка photos) -------------------
    const photoFiles = ["1.jpg", "2.jpg", "3.jpg", "4.jpg", "5.jpg"];
    function updatePhotoByDay(dayNumber) {
        if (!photoFiles.length) return;
        const idx = dayNumber % photoFiles.length;
        const name = photoFiles[idx];
        const img = document.getElementById('dailyPhoto');
        const cap = document.getElementById('photoCaption');
        img.src = "photos/" + name;
        cap.innerHTML = `📸 ${name.replace(/\.[^/.]+$/,"")} ・ день ${dayNumber}`;
        img.onerror = () => cap.innerHTML = `❌ не найден photos/${name}`;
    }

    // ===================== НЕПРЕРЫВНАЯ КАРУСЕЛЬ (requestAnimationFrame) =====================
    let photosArray = [];      // массив URL-адресов загруженных фото
    let frontTrack, bgTrack;
    let animationId = null;
    let offsetX = 0;           // текущее смещение в пикселях
    let speed = 1.2;          // пикселей за кадр (при 60fps ~72px/сек)
    let isAnimating = false;
    let lastTimestamp = 0;
    let cardWidthCached = 210;  // ширина карточки + gap (пересчитывается)
    let totalWidth = 0;

    // Создание DOM-элементов для карусели (передний и задний слои)
    function buildCarousel() {
        if (!frontTrack || !bgTrack) return;
        frontTrack.innerHTML = '';
        bgTrack.innerHTML = '';
        if (!photosArray.length) {
            const emptyMsg = document.createElement('div');
            emptyMsg.style.padding = '40px';
            emptyMsg.style.color = '#c0395e';
            emptyMsg.innerText = 'Загрузите фото через кнопку ниже ❤️';
            frontTrack.appendChild(emptyMsg);
            return;
        }
        // Для непрерывности создаём много копий (чтобы хватило на долгое движение без перестроения)
        const copies = 6;  // 6 полных копий массива
        for (let i = 0; i < copies; i++) {
            photosArray.forEach(src => {
                const cardFront = createCard(src);
                const cardBg = createCard(src);
                frontTrack.appendChild(cardFront);
                bgTrack.appendChild(cardBg);
            });
        }
        // Кэшируем ширину карточки с gap
        updateCardWidth();
        // Сбрасываем смещение
        offsetX = 0;
        if (frontTrack) frontTrack.style.transform = `translateX(0px)`;
        if (bgTrack) bgTrack.style.transform = `translateX(0px)`;
        // Запускаем анимацию
        startAnimation();
    }

    function createCard(src) {
        const card = document.createElement('div');
        card.className = 'carousel-card';
        const img = document.createElement('img');
        img.src = src;
        img.alt = 'воспоминание';
        img.onerror = () => { img.src = "https://placehold.co/400x400/ffb6c1/white?text=❤️"; };
        card.appendChild(img);
        return card;
    }

    function updateCardWidth() {
        const firstCard = document.querySelector('#carouselTrackFront .carousel-card');
        if (firstCard) {
            const rect = firstCard.getBoundingClientRect();
            const style = getComputedStyle(firstCard);
            const ml = parseFloat(style.marginLeft) || 0;
            const mr = parseFloat(style.marginRight) || 0;
            const parentGap = 16; // gap между карточками в .carousel-track
            cardWidthCached = rect.width + ml + mr + parentGap;
        } else {
            cardWidthCached = 216;
        }
        // общая ширина одного набора (одной копии массива)
        if (photosArray.length) {
            totalWidth = cardWidthCached * photosArray.length;
        }
    }

    // Обновление активного класса (центральное фото)
    function updateActiveClass() {
        if (!frontTrack) return;
        const cards = frontTrack.querySelectorAll('.carousel-card');
        if (!cards.length) return;
        const containerRect = document.querySelector('.carousel-wrapper').getBoundingClientRect();
        const centerX = containerRect.left + containerRect.width / 2;
        let closestIdx = -1;
        let minDist = Infinity;
        cards.forEach((card, idx) => {
            const rect = card.getBoundingClientRect();
            const cardCenter = rect.left + rect.width / 2;
            const dist = Math.abs(centerX - cardCenter);
            if (dist < minDist) {
                minDist = dist;
                closestIdx = idx;
            }
        });
        cards.forEach(c => c.classList.remove('active'));
        if (closestIdx !== -1) cards[closestIdx].classList.add('active');
    }

    // Анимация движения
    function animate(timestamp) {
        if (!isAnimating) return;
        // сдвигаем на скорость
        offsetX -= speed;
        // если ушли слишком далеко (на полную ширину одного набора), переставляем блоки
        if (Math.abs(offsetX) >= totalWidth && totalWidth > 0) {
            // Сдвигаем на один полный набор карточек вперёд
            const shift = Math.floor(Math.abs(offsetX) / totalWidth) * totalWidth;
            offsetX += shift;
            // Переставляем первые N карточек в конец (для двух треков)
            const frontCards = frontTrack.children;
            const bgCards = bgTrack.children;
            const numToMove = Math.floor(shift / cardWidthCached);
            if (numToMove > 0 && frontCards.length) {
                for (let i = 0; i < numToMove; i++) {
                    if (frontCards[i]) frontTrack.appendChild(frontCards[i]);
                    if (bgCards[i]) bgTrack.appendChild(bgCards[i]);
                }
            }
        }
        // применяем трансформацию
        frontTrack.style.transform = `translateX(${offsetX}px)`;
        bgTrack.style.transform = `translateX(${offsetX}px)`;
        // обновляем активный класс
        updateActiveClass();
        // продолжаем анимацию
        animationId = requestAnimationFrame(animate);
    }

    function startAnimation() {
        if (animationId) cancelAnimationFrame(animationId);
        if (photosArray.length > 0) {
            isAnimating = true;
            animationId = requestAnimationFrame(animate);
        }
    }

    function stopAnimation() {
        if (animationId) {
            cancelAnimationFrame(animationId);
            animationId = null;
        }
        isAnimating = false;
    }

    // Загрузка новых фото
    function addPhotoFiles(files) {
        if (!files.length) return;
        const newUrls = [];
        Array.from(files).forEach(file => {
            if (!file.type.startsWith('image/')) return;
            const url = URL.createObjectURL(file);
            newUrls.push(url);
        });
        if (newUrls.length) {
            photosArray.push(...newUrls);
            if (modal.style.display === 'flex') {
                stopAnimation();
                buildCarousel();
            } else {
                buildCarousel();
            }
        }
    }

    function clearAllPhotos() {
        if (confirm('Удалить все воспоминания?')) {
            photosArray.forEach(url => URL.revokeObjectURL(url));
            photosArray = [];
            if (modal.style.display === 'flex') {
                stopAnimation();
                buildCarousel();
            } else {
                buildCarousel();
            }
        }
    }

    // ------------------- МОДАЛЬНОЕ ОКНО -------------------
    const modal = document.getElementById('memoriesModal');
    const memoriesBtn = document.getElementById('memoriesBtn');
    const closeModalBtn = document.getElementById('closeModalBtn');
    const uploadBtn = document.getElementById('uploadMemoriesBtn');
    const clearBtn = document.getElementById('clearMemoriesBtn');

    memoriesBtn.onclick = () => {
        modal.style.display = 'flex';
        // инициализация треков
        frontTrack = document.getElementById('carouselTrackFront');
        bgTrack = document.getElementById('carouselTrackBg');
        buildCarousel();
        // пересчёт ширины каждые 2 секунды на случай изменения размеров окна
        const resizeObserver = () => {
            updateCardWidth();
            if (photosArray.length) startAnimation();
        };
        window.addEventListener('resize', resizeObserver);
        setTimeout(() => updateActiveClass(), 100);
        // сохраним функцию для удаления
        modal._resizeHandler = resizeObserver;
    };

    function closeModal() {
        modal.style.display = 'none';
        stopAnimation();
        if (modal._resizeHandler) {
            window.removeEventListener('resize', modal._resizeHandler);
            delete modal._resizeHandler;
        }
    }
    closeModalBtn.onclick = closeModal;
    window.onclick = (e) => { if (e.target === modal) closeModal(); };

    // Загрузка фото через input
    const fileInput = document.createElement('input');
    fileInput.type = 'file';
    fileInput.multiple = true;
    fileInput.accept = 'image/*';
    uploadBtn.onclick = () => {
        fileInput.value = '';
        fileInput.click();
    };
    fileInput.onchange = (e) => {
        if (fileInput.files && fileInput.files.length) addPhotoFiles(fileInput.files);
    };
    clearBtn.onclick = clearAllPhotos;

    // ------------------- АВТООБНОВЛЕНИЕ СЧЁТЧИКА В ПОЛНОЧЬ -------------------
    function scheduleDailyUpdate() {
        const now = new Date();
        const midnight = new Date();
        midnight.setHours(24, 0, 0, 0);
        const ms = midnight - now;
        setTimeout(() => {
            const days = updateCounter();
            updatePhotoByDay(days);
            setInterval(() => {
                const newDays = updateCounter();
                updatePhotoByDay(newDays);
            }, 24 * 60 * 60 * 1000);
        }, ms);
    }

    // ------------------- ЗАПУСК -------------------
    updateCounter();
    scheduleDailyUpdate();
    updatePhotoByDay(getDaysTogether());
    // инициализация пустой карусели
    frontTrack = document.getElementById('carouselTrackFront');
    bgTrack = document.getElementById('carouselTrackBg');
    if (frontTrack) buildCarousel();
</script>
